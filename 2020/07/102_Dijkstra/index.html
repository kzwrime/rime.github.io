Dijkstra BFS DFS 最短路径

Dijkstra BFS DFS 最短路径

本文用 c 实现 DFS, BFS, Dijkstra 寻找最短路径的算法。伪码和图文说明可以查看本博客的 另一篇文章。

理论上BFS和DFS找到的只是一条路径，但加以改造(加上更新路程)也是可以找最短路径的。此外除了数据结构不同，其实我的三种实现几乎一样 。。。

本文的一些数据结构的函数写着写着没用到，那就放那里吧，数据结构建议不看，如有需要可以点击下面各个小三角形查看折叠的数据结构代码。

/* 链表 */
typedef struct chainnode {
    int element;
    int weight;
    struct chainnode* next;
} ChainNode;

typedef struct chain {
    ChainNode* first;
    ChainNode* last;
} chain;

ChainNode* chainBegin(chain* ch);
ChainNode* chainEnd(chain* ch);
chain* createChain();
int emptyChain(chain* ch);
int insertChain(chain* ch, int data, int weight);
int chainErase(chain* ch, int data);
int deleteChain(chain* ch);

int getEleChain(chain* ch, int index);
int getWeight(chain* ch, int data);
chain* read(FILE* fp, int vv, int ww);
int deread(chain* lnn, int vv);
int popChain(chain* lnn);
int pushChain(chain* lnn, int data);

ChainNode* chainBegin(chain* ch)
{
    return ch->first;
}

ChainNode* chainEnd(chain* ch)
{
    return ch->last;
}

chain* createChain()
{
    chain* newChain = (chain*)malloc(sizeof(chain));
    newChain->first = NULL;
    newChain->last = NULL;
    return newChain;
}

inline int insertChain(chain* ch, int data, int weight)
{
    ChainNode* newNode = (ChainNode*)malloc(sizeof(ChainNode));
    newNode->weight = weight;
    newNode->element = data;
    newNode->next = NULL;
    if (NULL == ch->first) {
        ch->first = newNode;
        ch->last = newNode;
        return 0;
    } else {
        ch->last->next = newNode;
        ch->last = newNode;
    }
    return 0;
}

int chainErase(chain* ch, int data)
{
    ChainNode* node = ch->first;
    if (node->element == data) {
        ch->first = node->next;
        free(node);
        node == NULL;
        return data;
    }
    ChainNode* prenode = node;
    node = node->next;
    while (node != NULL) {
        if (node->element == data) {
            if (ch->last == node) {
                ch->last = prenode;
            }
            prenode->next = node->next;
            free(node);
            node == NULL;
            return data;
        }
        prenode = node;
        node = node->next;
    }
    return -1;
}

int deleteChain(chain* ch)
{
    if (NULL == ch->first) {
        return 0;
    }
    ChainNode* node = ch->first;
    ChainNode* nextNode = node->next;
    while (nextNode != NULL) {
        free(node);
        node = nextNode;
        nextNode = node->next;
    }
    free(node);
    free(ch);
    ch = NULL;
    return 0;
}

int getEleChain(chain* ch, int index)
{
    if (NULL == ch->first) {
        return -1;
    }

    int i = 0;
    ChainNode* node = ch->first;
    while (node != NULL) {
        if (i == index) {
            return node->element;
        }
        i++;
        node = node->next;
    }
    return -1;
}

inline int getWeight(chain* ch, int data)
{
    ChainNode* node = ch->first;
    while (node != NULL) {
        if (node->element == data) {
            return node->weight;
        }
        node = node->next;
    }
    return 0;
}

chain* read(FILE* fp, int vv, int ww)
{
    int a[3];
    chain* ln = (chain*)malloc(vv * sizeof(chain));
    for (int i = 0; i < vv; i++) {
        ln[i].first = NULL;
        ln[i].last = NULL;
    }
    printf("begin scaning....\n");
    while (fscanf(fp, "%d %d %d", &a[0], &a[1], &a[2]) != EOF) {
        insertChain(&(ln[a[0]]), a[1], a[2]);
        // insertChain(&(ln[a[1]]), a[0], a[2]); // 变为无向图，做弱连通测试
    }
    // *lnn = ln;
    // printf("ln=%d\n",ln);
    return ln;
}

int deread(chain* lnn, int vv)
{
    for (int i = 0; i < vv; i++) {
        deleteChain(&(lnn[i]));
    }
    return 0;
}

int popChain(chain* lnn)
{
    if (NULL == lnn->first) {
        return -1;
    }
    int data = lnn->first->element;
    ChainNode* old = lnn->first;
    lnn->first = lnn->first->next;
    free(old);
    return data;
}

int pushChain(chain* lnn, int data)
{
    ChainNode* newNode;
    newNode = (ChainNode*)malloc(sizeof(ChainNode));
    newNode->element = data;
    newNode->next = lnn->first;
    if (NULL == lnn->first) {
        lnn->last = newNode;
    }
    lnn->first = newNode;
    return 0;
}

优先队列（小根堆） 
/* 优先队列（小根堆） */
typedef struct {
    int* heap;
    int arrayLength;
    int heapSize;
} minHeap;
minHeap* minHeapCreate(int theSize);
int popMinHeap(minHeap* mhp, int* disFromSource);
void pushMinHeap(minHeap* mhp, int data, int* disFromSource);
void minHeapDestory(minHeap* mhp);

// 注意堆的序号，从1开始！！！

minHeap* minHeapCreate(int theSize)
{
    minHeap* mhp;
    mhp = (minHeap*)malloc(sizeof(minHeap));
    mhp->heapSize = 0;
    mhp->arrayLength = theSize;
    mhp->heap = (int*)malloc(theSize * sizeof(int));
    return mhp;
}
inline int popMinHeap(minHeap* mhp, int* disFromSource)
{
    if (0 == mhp->heapSize) {
        return -1;
    }

    int re = mhp->heap[1];

    int lastElement = mhp->heap[mhp->heapSize--]; // 先运算后自减

    // 为 lastElement 寻找位置
    int currentNode = 1;
    int child = 2;
    while (child <= mhp->heapSize) {
        //  mhp->heap[child] 应该是 currentNode 更小的孩子
        if (child < mhp->heapSize && disFromSource[mhp->heap[child]] > disFromSource[mhp->heap[child + 1]]) {
            child++;
        }

        if (disFromSource[lastElement] <= disFromSource[mhp->heap[currentNode]]) {
            break;
        }

        mhp->heap[currentNode] = mhp->heap[child];
        currentNode = child;
        child *= 2;
    }
    mhp->heap[currentNode] = lastElement;
    return re;
}
inline void pushMinHeap(minHeap* mhp, int data, int* disFromSource)
{
    // 适当时增加堆的数组长度
    if (mhp->heapSize == mhp->arrayLength - 1) {
        int* newHeap;
        newHeap = (int*)malloc(mhp->arrayLength * 2 * sizeof(int));
        for (int i = 0; i < mhp->arrayLength; i++) {
            newHeap[i] = mhp->heap[i];
        }
        free(mhp->heap);
        mhp->heap = newHeap;
        mhp->arrayLength *= 2;
        printf("%d\n", mhp->arrayLength);
    }

    // 为 data 寻找位置
    mhp->heapSize++;
    int currentNode = mhp->heapSize;
    while (currentNode != 1 && disFromSource[mhp->heap[currentNode / 2]] > disFromSource[data]) {
        // 和父节点互换
        mhp->heap[currentNode] = mhp->heap[currentNode / 2];
        currentNode /= 2;
    }
    mhp->heap[currentNode] = data;
}
void minHeapDestory(minHeap* mhp)
{
    free(mhp->heap);
    mhp->heap = NULL;
    free(mhp);
    mhp = NULL;
}
/* array2d 是链表数组，vv 是点的总数，source为起点, preNode和disFromSource为输出 */
int dijkstra(chain* array2d, int vv, int source, int* preNode, int* disFromSource)
{
    chain* ln = array2d;
    minHeap* newReachableNode;  // 小根堆实现的优先队列
    newReachableNode = minHeapCreate((vv / 100 > 20) ? vv / 100 : 20);

    /* 所有和起点相邻的点的距离，设置他们的前一个点为起点，并全部压入优先队列 */
    for (int i = 0; i < vv; i++) {
        disFromSource[i] = 0;
        preNode[i] = -1;
    }
    ChainNode* it = array2d[source].first;
    while (it != NULL) {
        disFromSource[it->element] = it->weight;
        preNode[it->element] = source;
        pushMinHeap(newReachableNode, it->element, disFromSource);
        it = it->next;
    }
    disFromSource[source] = 0;
    preNode[source] = -2;

    int v;
    while (0 != newReachableNode->heapSize) {
        v = popMinHeap(newReachableNode, disFromSource);    // 弹出目前距离最小的

        /* 更新所有和 v 相邻， 没到过或者距离可以更小的点，且没到过则入优先队列 */
        ChainNode* iterator = array2d[v].first;
        while (iterator != NULL) {
            if ((preNode[iterator->element] == -1 || 
                disFromSource[iterator->element] > disFromSource[v] + iterator->weight)) {
                disFromSource[iterator->element] = disFromSource[v] + iterator->weight;
                if (-1 == preNode[iterator->element]) {
                    pushMinHeap(newReachableNode, iterator->element, disFromSource);
                }
                preNode[iterator->element] = v;
            }
            iterator = iterator->next;
        }
    }
    minHeapDestory(newReachableNode);
    return 0;
}

/* 队列 */
typedef struct arrayQueue {
    int nOfQueue;
    int* array;
    int writePointer, readPointer;
    int free;
} ArrayQueue;

ArrayQueue* aCQ_create(int nOfQueue);
void aCQ_delete(ArrayQueue* aCQ);
int aCQ_empty(ArrayQueue* aCQ);
int enQueue(ArrayQueue* aCQ, int var);
int deQueue(ArrayQueue* aCQ);
int headOfQueue(ArrayQueue* aCQ);

ArrayQueue* aCQ_create(int nOfQueue)
{
    ArrayQueue* aCQ = (ArrayQueue*)malloc(sizeof(ArrayQueue));
    aCQ->nOfQueue = nOfQueue;
    aCQ->free = nOfQueue;
    aCQ->writePointer = 0;
    aCQ->readPointer = 0;
    aCQ->array = (int*)malloc(nOfQueue * sizeof(int));
    return aCQ;
}

void aCQ_delete(ArrayQueue* aCQ)
{
    free(aCQ->array);
    aCQ->array = NULL;
    free(aCQ);
    aCQ = NULL;
}

int aCQ_empty(ArrayQueue* aCQ)
{
    if (aCQ->free == aCQ->nOfQueue) {
        return 1;
    }
    return 0;
}

int enQueue(ArrayQueue* aCQ, int var)
{
    if (0 == aCQ->free) {
        return -1;
    }
    aCQ->array[aCQ->writePointer] = var;
    // aCQ->writePointer=(aCQ->writePointer+1)%(aCQ->nOfQueue);
    aCQ->writePointer++;
    aCQ->free--;
    return 0;
}

int deQueue(ArrayQueue* aCQ)
{
    if (aCQ->free == aCQ->nOfQueue) {
        return -1;
    }
    int data = aCQ->array[aCQ->readPointer];
    // aCQ->readPointer=(aCQ->readPointer+1)%(aCQ->nOfQueue);
    aCQ->readPointer++;
    aCQ->free++;
    return data;
}

int headOfQueue(ArrayQueue* aCQ)
{
    return aCQ->array[aCQ->readPointer];
}
void bfs(chain* array2d, int vv, int source, int* preNode, int* disFromSource)
{
    ArrayQueue* queue = aCQ_create(vv);

    /* 所有和起点相邻的点的距离，设置他们的前一个点为起点 */
    for (int i = 0; i < vv; i++) {
        disFromSource[i] = 0;
        preNode[i] = -1;
    }
    ChainNode* it = array2d[source].first;
    while (it != NULL) {
        disFromSource[it->element] = it->weight;
        preNode[it->element] = source;
        it = it->next;
    }
    disFromSource[source] = 0;
    preNode[source] = -2;

    enQueue(queue, source);

    while (!aCQ_empty(queue)) {
        int v = deQueue(queue);

        /* 更新所有和 v 相邻， 没到过或者距离可以更小的点，且没到过则入队列 */
        ChainNode* iterator = array2d[v].first;
        while (iterator != NULL) {
            if ((preNode[iterator->element] == -1 || disFromSource[iterator->element] > disFromSource[v] + iterator->weight)) {
                disFromSource[iterator->element] = disFromSource[v] + iterator->weight;
                enQueue(queue, iterator->element);  // 其实队列中可能重复，但问题不大
                preNode[iterator->element] = v;
            }
            iterator = iterator->next;
        }
    }
    // printf("success!\n");
    preNode[source] = -2;
    aCQ_delete(queue);
}

/* 栈 */
typedef struct arrayStack {
    int nOfArray;
    int* array;
    int pointer;
} ArrayStack;

ArrayStack* aS_create(int nOfArray);
void aS_delete(ArrayStack* aS);
int empty(ArrayStack* aS);
int push(ArrayStack* aS, int var);
int pop(ArrayStack* aS);

ArrayStack* aS_create(int nOfArray)
{
    ArrayStack* aS = (ArrayStack*)malloc(sizeof(ArrayStack));
    aS->nOfArray = nOfArray;
    aS->pointer = 0;
    aS->array = (int*)malloc(nOfArray * sizeof(int));
    return aS;
}

void aS_delete(ArrayStack* aS)
{
    free(aS->array);
    aS->array = NULL;
    free(aS);
    aS = NULL;
}

inline int empty(ArrayStack* aS)
{
    if (0 == aS->pointer) {
        return 1;
    }
    return 0;
}

inline int push(ArrayStack* aS, int var)
{
    if (aS->pointer < (aS->nOfArray - 1)) {
        aS->array[aS->pointer] = var;
        aS->pointer += 1;
        return 0;
    }
    return -1;
}

inline int pop(ArrayStack* aS)
{
    if (!empty(aS)) {
        aS->pointer -= 1;
        return aS->array[aS->pointer];
    }
    return -1;
}
void dfs(chain* array2d, int vv, int source, int* preNode, int* disFromSource)
{
    ArrayStack* stack = aS_create(vv);
    push(stack, source);

    /* 所有和起点相邻的点的距离，设置他们的前一个点为起点 */
    for (int i = 0; i < vv; i++) {
        disFromSource[i] = 0;
        preNode[i] = -1;
    }
    ChainNode* it = array2d[source].first;
    while (it != NULL) {
        disFromSource[it->element] = it->weight;
        preNode[it->element] = source;
        it = it->next;
    }
    disFromSource[source] = 0;
    preNode[source] = -2;

    while (!empty(stack)) {
        int v = pop(stack);

        /* 更新所有和 v 相邻， 没到过或者距离可以更小的点，且没到过则入栈 */
        ChainNode* iterator = array2d[v].first;
        while (iterator != NULL) {
            if ((preNode[iterator->element] == -1 || disFromSource[iterator->element] > disFromSource[v] + iterator->weight)) {
                disFromSource[iterator->element] = disFromSource[v] + iterator->weight;
                push(stack, iterator->element);
                preNode[iterator->element] = v;
            }
            iterator = iterator->next;
        }
    }
    preNode[source] = -2;
    aS_delete(stack);
}
